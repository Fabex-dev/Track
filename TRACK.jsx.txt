const getWeekRange = (date=new Date()) => {
  const d = new Date(date);
  const day = d.getDay();
  const start = new Date(d); start.setDate(d.getDate()-day);
  const end   = new Date(d); end.setDate(d.getDate()+(6-day));
  return { start: start.toISOString().split("T")[0], end: end.toISOString().split("T")[0] };
};

const fmt = (n, code="NGN") => {
  const c = CURRENCIES.find(x=>x.code===code)||CURRENCIES[0];
  return `${c.symbol}${Number(n).toLocaleString(c.locale,{minimumFractionDigits:2,maximumFractionDigits:2})}`;
};
const fmtShort = (n, code="NGN") => {
  const c = CURRENCIES.find(x=>x.code===code)||CURRENCIES[0];
  if(n>=1_000_000) return `${c.symbol}${(n/1_000_000).toFixed(1)}M`;
  if(n>=1_000)     return `${c.symbol}${(n/1_000).toFixed(1)}k`;
  return `${c.symbol}${Number(n).toFixed(0)}`;
};

// ─── LOCAL STORAGE HOOK ───────────────────────────────────────────────────────
const usePersistedState = (key, defaultVal) => {
  const [state, setState] = useState(() => {
    try {
      const stored = localStorage.getItem(STORAGE_KEY);
      if (stored) {
        const all = JSON.parse(stored);
        if (all[key] !== undefined) return all[key];
      }
    } catch {}
    return defaultVal;
  });

  const save = useCallback((val) => {
    setState(prev => {
      const next = typeof val === "function" ? val(prev) : val;
      try {
        const stored = localStorage.getItem(STORAGE_KEY);
        const all = stored ? JSON.parse(stored) : {};
        all[key] = next;
        localStorage.setItem(STORAGE_KEY, JSON.stringify(all));
      } catch {}
      return next;
    });
  }, [key]);

  return [state, save];
};

// ─── PIE CHART (full circle, not donut) ──────────────────────────────────────
const PieChart = ({ segs, currency, size=180 }) => {
  const total = segs.reduce((s,c)=>s+c.value,0);
  const cx=size/2, cy=size/2, r=size/2-2;
  if(!total) return (
    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
      <circle cx={cx} cy={cy} r={r} fill="rgba(255,255,255,0.04)" stroke="rgba(255,255,255,0.08)" strokeWidth="1"/>
      <text x={cx} y={cy+5} textAnchor="middle" fontSize="11" fill="rgba(255,255,255,0.25)" fontFamily="monospace">no data</text>
    </svg>
  );

  let angle = -Math.PI/2;
  const paths = segs.filter(s=>s.value>0).map((s,i) => {
    const slice = (s.value/total)*2*Math.PI;
    const x1 = cx + r*Math.cos(angle);
    const y1 = cy + r*Math.sin(angle);
    angle += slice;
    const x2 = cx + r*Math.cos(angle);
    const y2 = cy + r*Math.sin(angle);
    const large = slice>Math.PI?1:0;
    // label position
    const mid = angle - slice/2;
    const lx = cx + (r*0.65)*Math.cos(mid);
    const ly = cy + (r*0.65)*Math.sin(mid);
    const pct = Math.round((s.value/total)*100);
    return { d:`M${cx},${cy} L${x1},${y1} A${r},${r} 0 ${large},1 ${x2},${y2} Z`, color:s.color, lx, ly, pct, icon:s.icon };
  });

  return (
    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
      <defs>
        <filter id="ps"><feDropShadow dx="0" dy="2" stdDeviation="3" floodOpacity="0.4"/></filter>
      </defs>
      {paths.map((p,i)=>(
        <g key={i}>
          <path d={p.d} fill={p.color} stroke="rgba(8,12,24,0.9)" strokeWidth="1.5" filter="url(#ps)"/>
          {p.pct>=6&&(
            <text x={p.lx} y={p.ly+4} textAnchor="middle" fontSize="9.5" fill="rgba(0,0,0,0.85)" fontFamily="monospace" fontWeight="bold">{p.pct}%</text>
          )}
        </g>
      ))}
      {/* center circle overlay for glass effect */}
      <circle cx={cx} cy={cy} r={r*0.28} fill="rgba(8,12,24,0.92)" stroke="rgba(255,255,255,0.1)" strokeWidth="1"/>
      <text x={cx} y={cy-5} textAnchor="middle" fontSize="8" fill="rgba(255,255,255,0.4)" fontFamily="monospace">SPENT</text>
      <text x={cx} y={cy+8} textAnchor="middle" fontSize="10" fill="white" fontFamily="monospace" fontWeight="bold">
        {fmtShort(total,currency)}
      </text>
    </svg>
  );
};

// ─── DONUT CHART ─────────────────────────────────────────────────────────────
const Donut = ({ segs, currency }) => {
  const total = segs.reduce((s,c)=>s+c.value,0);
  if(!total) return (
    <svg viewBox="0 0 120 120" width="108" height="108">
      <circle cx="60" cy="60" r="40" fill="none" stroke="rgba(255,255,255,0.06)" strokeWidth="17"/>
      <text x="60" y="65" textAnchor="middle" fontSize="9" fill="rgba(255,255,255,0.2)" fontFamily="monospace">empty</text>
    </svg>
  );
  let off=0; const circ=2*Math.PI*40;
  return (
    <svg viewBox="0 0 120 120" width="108" height="108">
      <circle cx="60" cy="60" r="40" fill="none" stroke="rgba(255,255,255,0.06)" strokeWidth="17"/>
      {segs.filter(s=>s.value>0).map((s,i)=>{
        const dash=(s.value/total)*circ, gap=circ-dash;
        const el=<circle key={i} cx="60" cy="60" r="40" fill="none" stroke={s.color} strokeWidth="17"
          strokeDasharray={`${dash} ${gap}`} strokeDashoffset={-off}
          style={{transform:"rotate(-90deg)",transformOrigin:"60px 60px"}}/>;
        off+=dash; return el;
      })}
      <text x="60" y="55" textAnchor="middle" fontSize="8" fill="rgba(255,255,255,0.35)" fontFamily="monospace">TOTAL</text>
      <text x="60" y="70" textAnchor="middle" fontSize="11" fill="white" fontFamily="monospace" fontWeight="bold">
        {fmtShort(total,currency)}
      </text>
    </svg>
  );
};

  const w=300,h=76,bw=Math.max(6,(w/data.length)-3);
  return (
    <svg width="100%" viewBox={`0 0 ${w} ${h}`} preserveAspectRatio="none" style={{display:"block"}}>
      <defs>
        <linearGradient id="bg1" x1="0" y1="0" x2="0" y2="1">
          <stop offset="0%" stopColor={acc} stopOpacity="0.9"/>
          <stop offset="100%" stopColor={acc} stopOpacity="0.25"/>
        </linearGradient>
      </defs>
      {data.map((d,i)=>{
        const bh=Math.max(2,(d.value/max)*(h-20)), x=(i/data.length)*w+1;
        return(<g key={i}>
          <rect x={x} y={h-bh-16} width={bw-1} height={bh} rx="4"

                  topCat={prevWeekData.segs[0]}
                />
              )}

              {/* THIS MONTH */}
              {thisMonthData.total>0&&(
                <ReportCard
                  title={`${MONTH_NAMES[now.getMonth()]} ${now.getFullYear()} — This Month`}
                  subtitle="Current month spending"
                  segs={thisMonthData.segs}
                  total={thisMonthData.total}
                  currency={currency}
                  acc={acc}
                  topCat={thisMonthData.segs[0]}
                />
              )}

              {/* LAST MONTH */}
              {prevMonthData.total>0&&(
                <ReportCard
                  title={`${prevMonthData.label} — Last Month`}
                  subtitle="Previous month breakdown"
                  segs={prevMonthData.segs}
                  total={prevMonthData.total}
                  currency={currency}
                  acc={acc}
                  topCat={prevMonthData.segs[0]}
                />
              )}

              {thisWeekData.total===0&&prevWeekData.total===0&&thisMonthData.total===0&&prevMonthData.total===0&&(
                <div className="empty" style={{padding:"80px 20px"}}>
                  No data yet.<br/>Start logging expenses to see your reports.
                </div>
              )}
            </div>
          )}

          {/* ── SETTINGS ── */}
          {view==="settings"&&(
            <div className="gl addcard">
              {/* Name */}
              <div className="ssec">
                <span className="sslabel">Your Name</span>
                {editName
                  ?<div style={{display:"flex",gap:8}}>
                    <input ref={nameRef} className="name-inp" defaultValue={userName} autoFocus/>
                    <button className="savebtn" onClick={()=>{setUserName(nameRef.current.value||"Friend");setEditName(false);showToast("Name saved ✓");}}>Save</button>
                  </div>
                  :<div style={{display:"flex",justifyContent:"space-between",alignItems:"center",background:"rgba(255,255,255,0.04)",borderRadius:12,padding:"12px 16px",border:"1px solid rgba(255,255,255,0.07)"}}>
                    <span style={{fontWeight:600}}>{userName}</span>
                    <button onClick={()=>setEditName(true)} style={{background:"none",border:"none",color:acc,fontSize:12,fontWeight:600,cursor:"pointer"}}>Edit</button>
                  </div>
                }
              </div>

              {/* Profile */}
              <div className="ssec">
                <span className="sslabel">Who Are You?</span>
                <div className="pgrid">
                  {PROFILES.map(p=>(
                    <button key={p.id} className={`pbtn${profile===p.id?" on":""}`} onClick={()=>switchProfile(p.id)}>
                      <div className="pbi">{p.icon}</div>
                      <div className="pbl">{p.label}</div>
                      <div className="pbd">{p.desc}</div>
                    </button>
                  ))}
                </div>
              </div>

              {/* Theme */}
              <div className="ssec">
                <span className="sslabel">App Theme</span>
                <div className="thgrid">
                  {THEMES.map(t=>(
                    <button key={t.id} className={`thbtn${themeId===t.id?" on":""}`}
                      style={{background:`linear-gradient(135deg,${t.a},${t.b})`}}
                      onClick={()=>setThemeId(t.id)}/>
                  ))}
                </div>
                <div style={{display:"flex",gap:6,marginTop:7}}>
                  {THEMES.map(t=>(
                    <div key={t.id} style={{flex:1,textAlign:"center",fontSize:9,fontFamily:"'DM Mono',monospace",color:themeId===t.id?"white":"rgba(255,255,255,0.28)"}}>{t.label}</div>
                  ))}
                </div>
              </div>

              {/* Currency */}
              <div className="ssec">
                <span className="sslabel">Currency</span>
                <div style={{display:"flex",flexWrap:"wrap",gap:8}}>
                  {CURRENCIES.map(c=>(
                    <button key={c.code}
                      onClick={()=>{setCurrency(c.code);showToast(`${c.flag} ${c.label}`);}}
                      style={{background:currency===c.code?`${acc}16`:"rgba(255,255,255,0.04)",border:`1.5px solid ${currency===c.code?acc:"rgba(255,255,255,0.07)"}`,borderRadius:10,padding:"8px 12px",cursor:"pointer",color:"white",fontFamily:"'DM Mono',monospace",fontSize:12,fontWeight:600,display:"flex",alignItems:"center",gap:5}}>
                      {c.flag} {c.symbol} {c.code}
                    </button>
                  ))}
                </div>
              </div>

              {/* Budgets */}
              <div className="ssec">
                <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:12}}>
                  <span className="sslabel" style={{margin:0}}>Custom Budgets</span>
                  {!editBudget&&<button onClick={()=>setEditBudget(true)} style={{background:"none",border:"none",color:acc,fontSize:12,fontWeight:700,cursor:"pointer"}}>Edit</button>}
                </div>
                {editBudget?(
                  <>
                    <div className="binputs">
                      {["daily","weekly","monthly"].map(p=>(
                        <div className="bfield" key={p}>
                          <label>{p}</label>
                          <input type="number" value={budget[p]} onChange={e=>setBudget(b=>({...b,[p]:Number(e.target.value)}))}/>
                        </div>
                      ))}
                    </div>
                    <button className="savebtn" onClick={()=>{setEditBudget(false);showToast("Budgets saved ✓");}}>Save</button>
                  </>
                ):(
                  <div style={{display:"flex",gap:8}}>
                    {["daily","weekly","monthly"].map(p=>(
                      <div key={p} style={{flex:1,background:"rgba(0,0,0,0.22)",borderRadius:12,padding:"12px 8px",textAlign:"center",border:"1px solid rgba(255,255,255,0.06)"}}>
                        <div style={{fontSize:9,color:"rgba(255,255,255,0.24)",textTransform:"uppercase",letterSpacing:1,marginBottom:4,fontFamily:"'DM Mono',monospace"}}>{p}</div>
                        <div style={{fontSize:14,fontWeight:700}}>{fmtShort(budget[p],currency)}</div>
                      </div>
                    ))}
                  </div>
                )}
              </div>

              {/* Danger zone */}
              <div className="ssec">
                <span className="sslabel">Data</span>
                <button onClick={()=>{if(window.confirm("Clear ALL expenses? This cannot be undone.")){setExpenses([]);showToast("All expenses cleared");}}} style={{background:"rgba(248,113,113,0.08)",border:"1px solid rgba(248,113,113,0.25)",borderRadius:12,padding:"11px 18px",color:"#f87171",fontFamily:"'Sora',sans-serif",fontSize:13,fontWeight:600,cursor:"pointer",width:"100%"}}>
                  🗑 Clear All Expenses
                </button>
              </div>

              {/* About */}
              <div style={{borderTop:"1px solid rgba(255,255,255,0.06)",paddingTop:20,textAlign:"center"}}>
                <TrackLogo size={36} acc={acc}/>
                <div style={{fontFamily:"'Sora',sans-serif",fontSize:20,fontWeight:800,letterSpacing:"-1px",marginTop:10}}>TRACK<span style={{color:acc}}>.</span></div>
                <div style={{fontFamily:"'DM Mono',monospace",fontSize:10,color:"rgba(255,255,255,0.2)",marginTop:4,letterSpacing:"2px"}}>crafted with ♥ by Fabex</div>
                <div style={{fontFamily:"'DM Mono',monospace",fontSize:9,color:"rgba(255,255,255,0.12)",marginTop:3,letterSpacing:"1px"}}>all data auto-saved locally</div>
              </div>
            </div>
          )}

        </div>

        {/* FAB — only on non-form views */}
        {["dashboard","history","calendar"].includes(view)&&(
          <button className="fab" onClick={()=>{setForm(f=>({...f,date:todayStr(),amount:"",note:""}));setView("add");}}>+</button>
        )}

        {/* ── BOTTOM NAV ── */}
        <div className="bnav">
          <button className={`nbtn${view==="dashboard"?" on":""}`} onClick={()=>setView("dashboard")}>
            <svg viewBox="0 0 24 24"><rect x="3" y="3" width="7" height="7" rx="1.5"/><rect x="14" y="3" width="7" height="7" rx="1.5"/><rect x="3" y="14" width="7" height="7" rx="1.5"/><rect x="14" y="14" width="7" height="7" rx="1.5"/></svg>
            <span>DASH</span>
          </button>
          <button className={`nbtn${view==="calendar"?" on":""}`} onClick={()=>setView("calendar")}>
            <svg viewBox="0 0 24 24"><rect x="3" y="4" width="18" height="18" rx="2"/><line x1="16" y1="2" x2="16" y2="6"/><line x1="8" y1="2" x2="8" y2="6"/><line x1="3" y1="10" x2="21" y2="10"/></svg>
            <span>CALENDAR</span>
          </button>
          <button className={`nbtn${view==="reports"?" on":""}`} onClick={()=>setView("reports")}>
            <svg viewBox="0 0 24 24"><circle cx="12" cy="12" r="10"/><path d="M12 2a10 10 0 0 1 10 10H12z"/><path d="M12 12 6.5 19.5"/></svg>
            <span>REPORTS</span>
          </button>
          <button className={`nbtn${view==="history"?" on":""}`} onClick={()=>setView("history")}>
            <svg viewBox="0 0 24 24"><line x1="8" y1="6" x2="21" y2="6"/><line x1="8" y1="12" x2="21" y2="12"/><line x1="8" y1="18" x2="21" y2="18"/><circle cx="3" cy="6" r="1.5" fill="currentColor" stroke="none"/><circle cx="3" cy="12" r="1.5" fill="currentColor" stroke="none"/><circle cx="3" cy="18" r="1.5" fill="currentColor" stroke="none"/></svg>
            <span>LOG</span>
          </button>
          <button className={`nbtn${view==="analysis"?" on":""}`} onClick={()=>setView("analysis")}>
            <svg viewBox="0 0 24 24"><polyline points="22 12 18 12 15 21 9 3 6 12 2 12"/></svg>
            <span>AI</span>
          </button>
          <button className={`nbtn${view==="settings"?" on":""}`} onClick={()=>setView("settings")}>
            <svg viewBox="0 0 24 24"><circle cx="12" cy="12" r="3"/><path d="M19.4 15a1.65 1.65 0 0 0 .33 1.82l.06.06a2 2 0 0 1-2.83 2.83l-.06-.06a1.65 1.65 0 0 0-1.82-.33 1.65 1.65 0 0 0-1 1.51V21a2 2 0 0 1-4 0v-.09A1.65 1.65 0 0 0 9 19.4a1.65 1.65 0 0 0-1.82.33l-.06.06a2 2 0 0 1-2.83-2.83l.06-.06A1.65 1.65 0 0 0 4.68 15a1.65 1.65 0 0 0-1.51-1H3a2 2 0 0 1 0-4h.09A1.65 1.65 0 0 0 4.6 9a1.65 1.65 0 0 0-.33-1.82l-.06-.06a2 2 0 0 1 2.83-2.83l.06.06A1.65 1.65 0 0 0 9 4.68a1.65 1.65 0 0 0 1-1.51V3a2 2 0 0 1 4 0v.09a1.65 1.65 0 0 0 1 1.51 1.65 1.65 0 0 0 1.82-.33l.06-.06a2 2 0 0 1 2.83 2.83l-.06.06A1.65 1.65 0 0 0 19.4 9a1.65 1.65 0 0 0 1.51 1H21a2 2 0 0 1 0 4h-.09a1.65 1.65 0 0 0-1.51 1z"/></svg>
            <span>SETTINGS</span>
          </button>
        </div>
      </div>
    </>
  );
}
